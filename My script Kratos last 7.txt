local invisible = false
local character, humanoid, root
local savedParts = {}

local function cacheParts()
    savedParts = {}
    for _,v in ipairs(character:GetDescendants()) do
        if v:IsA("BasePart") then
            table.insert(savedParts, v)
        end
    end
end

local function setInvisible(state)
    invisible = state
    for _,p in ipairs(savedParts) do
        p.Transparency = state and 0.5 or 0
    end
end

local function onCharacter(char)
    character = char
    humanoid = char:WaitForChild("Humanoid")
    root = char:WaitForChild("HumanoidRootPart")
    cacheParts()
end

local function GetMyProps()
    local props = {}
    local folder = workspace:FindFirstChild("WorkspaceCom")
    if not folder then return props end

    for _,f in pairs(folder:GetChildren()) do
        for _,p in pairs(f:GetChildren()) do
            if p.Name:find(LocalPlayer.Name) and p:FindFirstChild("SetCurrentCFrame") then
                table.insert(props,p)
            end
        end
    end

    return props
end

local function stopTracks(t1, t2)
    for _, t in pairs({t1, t2}) do

                t:Destroy()
            end)
        end

local function saveBody(desc, tbl)
    tbl.Torso = desc.Torso
    tbl.RightArm = desc.RightArm
    tbl.LeftArm = desc.LeftArm
    tbl.RightLeg = desc.RightLeg
    tbl.LeftLeg = desc.LeftLeg
    tbl.Head = desc.Head
end

local function restoreBody(tbl)
    if not next(tbl) then return end

    pcall(function()
        Rep.Remotes.ChangeCharacterBody:InvokeServer({
            tbl.Torso,
            tbl.RightArm,
            tbl.LeftArm,
            tbl.RightLeg,
            tbl.LeftLeg,
            tbl.Head
        })
    end)

    table.clear(tbl)
end

local function applyBody()
    pcall(function()
        Rep.Remotes.ChangeCharacterBody:InvokeServer({
            123402086843885,
            100839513065432,
            78300682916056,
            86276701020724,
            78409653958165,
            15093053680
        })
    end)
end

local function playAnim(id, speed, priority)
    local char = LocalPlayer.Character
    if not char then return end

    local hum = char:FindFirstChildOfClass("Humanoid")
    if not hum then return end

    local anim = Instance.new("Animation")
    anim.AnimationId = "rbxassetid://"..id

    local track = hum:LoadAnimation(anim)
    track.Priority = priority or Enum.AnimationPriority.Action
    track:Play()

    if speed then
        track:AdjustSpeed(speed)
    end

    return track
end

-- Noclip
local noclipEnabled = false

local function setCharacterCanCollide(character, canCollide)
    for _, part in ipairs(character:GetDescendants()) do
        if part:IsA("BasePart") then part.CanCollide = canCollide end
    end
end

    noclipEnabled = value
    local character = player.Character
    if value then
        noclipLoop = RunService.Stepped:Connect(function() if character then setCharacterCanCollide(character,false) end end)
    else
        if noclipLoop then noclipLoop:Disconnect() end
        if character then setCharacterCanCollide(character,true) end
    end
end})

local function spectatePlayer(enable)
    local camera = Workspace.CurrentCamera
    if selectedPlayer then
        local target = Players:FindFirstChild(selectedPlayer)
        if target and target.Character and target.Character:FindFirstChild("Humanoid") then
            camera.CameraSubject = enable and target.Character.Humanoid or player.Character.Humanoid
        end
    end
end

local connection = nil
local function floatAbovePlayerHead()
    local character = player.Character
    if not character or not selectedPlayer then return end
    local root = character:FindFirstChild("HumanoidRootPart")
    local target = Players:FindFirstChild(selectedPlayer)
    if root and target and target.Character and target.Character:FindFirstChild("Head") then
        if connection then connection:Disconnect() end
        connection = RunService.Heartbeat:Connect(function()
            root.CFrame = target.Character.Head.CFrame * CFrame.new(0,3,0)
        end)
    end
end
AdminTab:CreateButton({Name="اذهب الى الاعب", Callback=floatAbovePlayerHead})
